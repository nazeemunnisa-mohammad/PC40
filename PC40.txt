import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(home: ReorderListExample()));
}

class ReorderListExample extends StatefulWidget {
  @override
  _ReorderListExampleState createState() => _ReorderListExampleState();
}

class _ReorderListExampleState extends State<ReorderListExample> {
  // Initial list of items
  List<String> items = ['item1', 'item2', 'item3', 'item4', 'item5'];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Reorderable List Example")),
      body: ReorderableListView(
        onReorder: (int oldIndex, int newIndex) {
          setState(() {
            // Adjust newIndex if necessary
            if (newIndex > oldIndex) {
              newIndex -= 1;
            }
            final String movedItem = items.removeAt(oldIndex);
            items.insert(newIndex, movedItem);
          });
        },
        children: [
          for (final item in items)
            ListTile(
              key: ValueKey(item),
              title: Text(item, style: TextStyle(fontSize: 20)),
              leading: Icon(Icons.drag_handle),
            ),
        ],
      ),
    );
  }
}
